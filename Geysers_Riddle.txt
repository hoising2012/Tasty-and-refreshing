<?php
function set_Pr(int $n){
	$t="";
    switch($n){
    	case 3:
        $t=1/3;
        break;
        case 2:
        $t=1/2;
        break;
        case 1:
        $t=1;
        break;
    default:
    break;
    }
    return $t;
}
$erupt=array(array(1,1,1),
		array(1,1,0),
        array(1,0,1),
        array(1,0,0)
);
$probability=array();
$Pr=array();
for($i=0; $i<=3; $i++){
    $probability[$i]=1;
    for($j=0; $j<=2; $j++){
        $x=$erupt[$i][$j];
		if(($j==0)&&($x==1)){
        	$probability[$i]=$probability[$i]*1;
        }else if(($j==1)&&($x==1)){
			$probability[$i]=$probability[$i]*(1/2);
        }else if(($j==2)&&($x==1)){
        	$probability[$i]=$probability[$i]*(1/3); 
        }
       	if(($j==1)&&($x==0)){
        $probability[$i]=$probability[$i]*(1-1/2);
        }else if(($j==2)&&($x==0)){
        $probability[$i]=$probability[$i]*(1-1/3);
        }
	} 
}
for($i=0; $i<=3; $i++){
	$count = 0;
    for($j=0; $j<=2; $j++){
	if($erupt[$i][$j]==1){
    	$count++;
        }        
    }
    $Pr[$i]=set_Pr($count);
}
echo "<table>";

echo "<th>Pr(C|case)</th>";
echo "</tr>";
for($i=0; $i<=3; $i++){
	echo "<tr>";
    if($i==0){
    echo "<td>(I) A, B, C all erupt</td>";
    }if($i==1){
    echo "<td>(II) A, B erupt, C does not</td>";
    }if($i==2){
    echo "<td>(III) A, C erupt, B does not</td>";
    }if($i==3){
    echo "<td>(IV) A erupt, B, C does not</td>";
    }
	echo "<td>$probability[$i]</td>";
	for($j=0; $j<=2; $j++){
    	$x=$erupt[$i][$j];
  		if($x==1){
        	$m=$Pr[$i];
        }else if($x==0){
        $m=0;
        } 
        echo "<td>$m</td>";
    }
    echo "</tr>";
}
echo "</table>";
