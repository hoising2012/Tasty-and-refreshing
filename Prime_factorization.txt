$n=1092;

        echo "<tr>";

	echo "<tr>";

	$PC=primeCheck($i);
    if($PC==1){

        $PG=$n;
        $n=$t;
        $PF=$i;
    }

    echo "</tr>";

function primeCheck($number){
    if ($number == 1)
    return 0;
    for ($i = 2; $i <= $number/2; $i++){
        if ($number % $i == 0)
            return 0;
    }
    return 1;
}
